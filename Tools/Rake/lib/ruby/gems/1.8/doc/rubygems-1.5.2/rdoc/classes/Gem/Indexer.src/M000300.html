  def build_legacy_indicies(index)
    say "Generating Marshal master index"

    Gem.time 'Generated Marshal master index' do
      open @marshal_index, 'wb' do |io|
        io.write index.dump
      end
    end

    @files << @marshal_index
    @files << "#{@marshal_index}.Z"
  end
